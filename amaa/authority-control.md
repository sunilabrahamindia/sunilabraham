This page documents authority control identifiers, bibliographic records, scholarly profiles, archival references, and related metadata associated with [A. M. A. Ayrookuzhiel](/amaa/). It is intended to support long-term discoverability, citation consistency, digital preservation, and interoperability across library catalogues, theological archives, research databases, archival systems, and web-based scholarly infrastructure.

The page consolidates identifiers and external records connected with A. M. A. Ayrookuzhiel's publications, theological writings, sociological research, Dalit studies scholarship, editorial contributions, institutional affiliations, and historical work related to caste, religion, liberation theology, social reform, and social movements in Kerala and India.

As metadata systems frequently fragment identities across variant initials, transliterations, institutional records, and bibliographic databases, this page also serves as a reconciliation layer for authority control and bibliographic standardisation within The Sunil Abraham Project.

<script>
(function () {

  /* ── Search ── */
  var searchInput = document.getElementById('authority-search');
  var clearBtn    = document.getElementById('ac-clear');
  var countEl     = document.getElementById('ac-count');
  var emptyEl     = document.getElementById('ac-empty');
  var items       = document.querySelectorAll('.ac-item');
  var groups      = document.querySelectorAll('.ac-group-card');
  var TOTAL       = items.length;

  /* ── Parameter Preservation State Logic ── */
  function writeURL(q) {
    const currentParams = new URLSearchParams(window.location.search);
    const darkmodeValue = currentParams.get('darkmode');
    
    const params = new URLSearchParams();
    if (q) params.set('q', q);
    if (darkmodeValue !== null) params.set('darkmode', darkmodeValue);

    const newURL = params.toString()
      ? window.location.pathname + '?' + params.toString()
      : window.location.pathname;
    history.replaceState(null, '', newURL);
  }

  function readURL() {
    const params = new URLSearchParams(window.location.search);
    if (params.get('q')) {
      searchInput.value = params.get('q');
      runSearch(params.get('q').toLowerCase().trim(), false);
    }
  }

  function updateCount(n) {
    countEl.textContent = n + (n === 1 ? ' record' : ' records');
  }

  function runSearch(q, shouldUpdateURL = true) {
    var visible = 0;
    items.forEach(function (item) {
      var label = (item.dataset.label || '').toLowerCase();
      var value = (item.dataset.value || '').toLowerCase();
      var text  = item.textContent.toLowerCase();
      var match = !q || label.includes(q) || value.includes(q) || text.includes(q);
      item.setAttribute('aria-hidden', match ? 'false' : 'true');
      if (match) visible++;
    });

    /* Hide groups with no visible items */
    groups.forEach(function (grp) {
      var anyVisible = Array.from(grp.querySelectorAll('.ac-item'))
        .some(function (it) { return it.getAttribute('aria-hidden') !== 'true'; });
      grp.setAttribute('aria-hidden', anyVisible ? 'false' : 'true');
    });

    updateCount(q ? visible : TOTAL);
    emptyEl.classList.toggle('visible', visible === 0 && !!q);
    clearBtn.classList.toggle('visible', !!q);
    
    if (shouldUpdateURL) {
      writeURL(q);
    }
  }

  searchInput.addEventListener('input', function () {
    runSearch(this.value.toLowerCase().trim());
  });

  clearBtn.addEventListener('click', function () {
    searchInput.value = '';
    runSearch('');
    searchInput.focus();
  });

  /* Keyboard: Escape clears */
  searchInput.addEventListener('keydown', function (e) {
    if (e.key === 'Escape' && this.value) {
      this.value = '';
      runSearch('');
    }
  });

  updateCount(TOTAL);
  readURL();
})();
</script>
