The [Centre for Internet and Society](/cis/) (CIS) has maintained a Board and a Society since its establishment in 2008. This page presents an interactive visualisation of the organisation's Board and Society membership from the 2008–09 to the 2023–24 financial year, based on the official membership lists published by CIS.

The visualisations summarise changes in membership, leadership positions, years of service and institutional continuity over this sixteen-year period. Rather than reproducing the original tabular records, they present the information through timelines, charts and other graphical representations intended to facilitate historical exploration, comparison and analysis.

**Note:** This page follows the official CIS webpage, **[Board and Society Members](https://cis-india.org/about/people/board-and-society-members)**. The official page provides membership information up to the 2023–24 financial year. Accordingly, the visualisations and statistics presented here are limited to the period from 2008–09 to 2023–24.

  <details>
    <summary>Data Processing Notes</summary>
    <div>
      <p>The dataset was parsed from the CIS "Board and Society Members" page, which lists members and positions for each financial year from 2008&ndash;09 to 2023&ndash;24. Each yearly table was transcribed into a structured JavaScript object keyed by year, with separate arrays for Board and Society entries, each holding a name and a position.</p>
      <p>Names were normalised by stripping honorific prefixes (Mr., Ms., Mrs., Dr., Prof.) so that the same individual is recognised consistently across years even where the source used inconsistent styling (for example, "Sunil Abraham" versus "Mr. Sunil Abraham"). One spelling inconsistency was detected and reconciled: "Subbaiah Arunachalam" (used in 2020&ndash;21 and 2021&ndash;22) was treated as the same person as "Subbiah Arunachalam" used elsewhere, since spellings, positions and years of activity align.</p>
      <p>Where the source page displayed sparse or seemingly incomplete role labels (for instance, 2019&ndash;20's Society list), the entries were transcribed as they appeared, without inferring missing roles or members. Where a "Chief Functionary" or "Chairperson/Chairman" role appeared, it was recorded as-is rather than mapped onto a fixed role taxonomy, since CIS's own titles changed over time.</p>
      <p>All charts, counters, the heatmap, the leadership timeline and the founding member cards are generated programmatically from this single structured dataset at page load; no chart values are hardcoded. Any future correction to the source data only requires updating the embedded dataset object.</p>
    </div>
  </details>

/* ===================== DATA PROCESSING ===================== */
const YEARS = Object.keys(RAW_DATA).sort();
const people = {};
YEARS.forEach(year => {
  ['board','society'].forEach(grp => {
    RAW_DATA[year][grp].forEach(([name,pos]) => {
      if(!people[name]) people[name] = {board:{}, society:{}};
      people[name][grp][year] = pos;
    });
  });
});

const validation = {duplicates:[], missingPositions:[], totalPeople:Object.keys(people).length, totalAppointments:0};
YEARS.forEach(year=>{
  ['board','society'].forEach(grp=>{
    RAW_DATA[year][grp].forEach(([name,pos])=>{
      validation.totalAppointments++;
      if(!pos) validation.missingPositions.push(`${name} (${year}, ${grp})`);
    });
  });
});

function personYears(p, grp){ return Object.keys(p[grp]).sort(); }
function personSummary(name){
  const p = people[name];
  const by = personYears(p,'board'), sy = personYears(p,'society');
  const allYears = Array.from(new Set([...by, ...sy])).sort();
  const first = allYears[0], last = allYears[allYears.length-1];
  const roles = [];
  allYears.forEach(y=>{
    if(p.board[y]) roles.push({year:y, grp:'Board', pos:p.board[y]});
    if(p.society[y]) roles.push({year:y, grp:'Society', pos:p.society[y]});
  });
  const leadershipRoles = roles.filter(r=>r.pos && r.pos!=='Member');
  return {name, first, last, totalYears:allYears.length, boardYears:by.length, societyYears:sy.length, allYears, roles, leadershipRoles};
}

const ALL_NAMES = Object.keys(people).sort();
const SUMMARIES = ALL_NAMES.map(personSummary);

let totalBoardApps=0, totalSocietyApps=0;
YEARS.forEach(y=>{ totalBoardApps += RAW_DATA[y].board.length; totalSocietyApps += RAW_DATA[y].society.length; });

/* ===================== ANIMATED COUNTERS ===================== */
function animateCounter(el, target, duration=1200){
  const start = performance.now();
  function tick(now){
    const p = Math.min(1, (now-start)/duration);
    el.textContent = Math.floor(p*target);
    if(p<1) requestAnimationFrame(tick); else el.textContent = target;
  }
  requestAnimationFrame(tick);
}
function initCounters(){
  animateCounter(document.getElementById('cisStatYears'), YEARS.length);
  animateCounter(document.getElementById('cisStatPeople'), ALL_NAMES.length);
  animateCounter(document.getElementById('cisStatBoardApps'), totalBoardApps);
  animateCounter(document.getElementById('cisStatSocietyApps'), totalSocietyApps);
  animateCounter(document.getElementById('cisStatFounders'), new Set([...META.foundingBoard, ...META.foundingSociety]).size);
}

/* ===================== HEATMAP ===================== */
function renderHeatmap(){
  const table = document.getElementById('cisHeatmapTable');
  const rows = [...SUMMARIES].sort((a,b)=> a.first.localeCompare(b.first) || b.totalYears-a.totalYears);
  let thead = '<thead><tr><th class="rowhead">Person</th>' + YEARS.map(y=>`<th scope="col">${y.slice(2,4)}&ndash;${y.slice(7,9)}</th>`).join('') + '</tr></thead>';
  let tbody = '<tbody>' + rows.map(s=>{
    const p = people[s.name];
    const cells = YEARS.map(y=>{
      const b = p.board[y], so = p.society[y];
      let cls='hm-none', label='Not a member';
      if(b && so){cls='hm-both'; label=`Board (${b}) & Society (${so})`;}
      else if(b){cls='hm-board'; label=`Board (${b})`;}
      else if(so){cls='hm-society'; label=`Society (${so})`;}
      return `<td><div class="hm-cell ${cls}" tabindex="${cls!=='hm-none'?'0':'-1'}" role="img" aria-label="${s.name}, ${y}: ${label}" title="${s.name} &middot; ${y} &middot; ${label}"></div></td>`;
    }).join('');
    return `<tr><th scope="row" class="rowhead">${personButton(s.name)}</th>${cells}</tr>`;
  }).join('') + '</tbody>';
  table.innerHTML = thead + tbody;
}

function themeColors(){
  const dark = window.matchMedia('(prefers-color-scheme: dark)').matches;
  return {
    text: dark ? '#eae7e1' : '#1a1a1a',
    grid: dark ? 'rgba(255,255,255,0.08)' : 'rgba(0,0,0,0.08)',
    board: dark ? '#e08a8a' : '#8a2c2c',
    society: dark ? '#6fb8cf' : '#1e4d5c'
  };
}

  Chart.defaults.font.family = "-apple-system,BlinkMacSystemFont,'Segoe UI',Roboto,Helvetica,Arial,sans-serif";

  });
}

function renderTransitions(){
  const el = document.getElementById('cisTransitionsView');
  const roleTrack = {};
  YEARS.forEach(year=>{
    ['board','society'].forEach(grp=>{
      RAW_DATA[year][grp].forEach(([name,pos])=>{
        if(LEADER_ROLES.has(pos)){
          const key = pos + ' (' + grp + ')';
          roleTrack[key] = roleTrack[key] || [];
          const last = roleTrack[key][roleTrack[key].length-1];
          if(!last || last.name !== name) roleTrack[key].push({year, name});
        }
      });
    });
  });
  el.innerHTML = Object.entries(roleTrack).map(([role, seq])=>{
    const chain = seq.map((s,i)=>`<span class="role-pill">${s.year}: ${personButton(s.name)}</span>${i<seq.length-1?' <span aria-hidden="true">&rarr;</span> ':''}`).join('');
    return `<div class="timeline-row"><div class="timeline-year" style="font-size:0.85rem;">${role}</div><div class="timeline-roles">${chain}</div></div>`;
  }).join('');
}

let currentFilter = 'all';
let currentQuery = '';
function matchesFilter(s){
  if(currentFilter==='board') return s.boardYears>0 && s.societyYears===0;
  if(currentFilter==='society') return s.societyYears>0 && s.boardYears===0;
  if(currentFilter==='both') return s.boardYears>0 && s.societyYears>0;
  if(currentFilter==='leadership') return s.leadershipRoles.length>0;
  return true;
}
function renderSearch(){
  const el = document.getElementById('cisSearchResults');
  const q = currentQuery.toLowerCase();
  const results = SUMMARIES.filter(s=> s.name.toLowerCase().includes(q) && matchesFilter(s));
  el.innerHTML = results.map(s=>`
    <div class="card">
      <h3>${personButton(s.name)}</h3>
      <p>${s.first}&ndash;${s.last} &middot; ${s.totalYears} year(s)</p>
      <p>Board: ${s.boardYears} &middot; Society: ${s.societyYears}${s.leadershipRoles.length? ' &middot; Leadership roles: '+s.leadershipRoles.length:''}</p>
    </div>`).join('') || '<p>No matching individuals found.</p>';
}

}

/* ===================== SCROLLSPY / NAV / COPY LINK / BACK TO TOP ===================== */
function initNav(){
  const links = document.querySelectorAll('.cis-viz nav.toc a');
  const sections = [...links].map(l=>document.querySelector(l.getAttribute('href')));
  function onScroll(){
    let idx = 0;
    sections.forEach((s,i)=>{ if(s && s.getBoundingClientRect().top < 120) idx = i; });
    links.forEach((l,i)=> l.classList.toggle('active', i===idx));
    const backBtn = document.getElementById('cisBackToTop');
    backBtn.classList.toggle('show', window.scrollY > 500);
  }
  document.addEventListener('scroll', onScroll, {passive:true});
  onScroll();
}

    setTimeout(()=> btn.textContent = 'Copy link', 1500);
