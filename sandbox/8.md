function uniq(arr) { return [...new Set(arr.filter(Boolean))].sort(); }
const allEventTypes = uniq(WQ_DATA.map(d => d.event_type));
const allLocations  = uniq(WQ_DATA.map(d => d.location));
const allOrganisers = uniq(WQ_DATA.flatMap(d => d.organisers));
const allRoles      = uniq(WQ_DATA.flatMap(d => d.sunils_role));

/* ── Tag input factory ── */
function makeTagInput(listEl, inputEl, suggestEl, pool) {
  const tags = [];
  function render() {
    const chips = listEl.querySelectorAll('.wq-chip');
    chips.forEach(c => c.remove());
    tags.forEach((t, i) => {
      const chip = document.createElement('span');
      chip.className = 'wq-chip';
      chip.innerHTML = `${t}<button aria-label="Remove ${t}">✕</button>`;
      chip.querySelector('button').addEventListener('click', () => { tags.splice(i,1); render(); });
      listEl.insertBefore(chip, inputEl);
    });
  }
  function add(v) {
    const t = (v||'').trim();
    if (t && !tags.includes(t)) { tags.push(t); render(); }
  }
  function suggest(q) {
    const ql = q.toLowerCase();
    const hits = pool.filter(x => x.toLowerCase().includes(ql) && !tags.includes(x)).slice(0,20);
    suggestEl.innerHTML = '';
    if (!hits.length) { suggestEl.style.display = 'none'; return; }
    hits.forEach(h => {
      const b = document.createElement('button');
      b.type = 'button'; b.textContent = h;
      b.addEventListener('click', () => { add(h); inputEl.value = ''; suggestEl.style.display = 'none'; });
      suggestEl.appendChild(b);
    });
    suggestEl.style.display = 'block';
  }
  inputEl.addEventListener('keydown', e => {
    if (e.key === 'Enter') { e.preventDefault(); add(inputEl.value); inputEl.value = ''; suggestEl.style.display = 'none'; }
    if (e.key === 'Escape') { inputEl.value = ''; suggestEl.style.display = 'none'; }
  });
  inputEl.addEventListener('input', () => inputEl.value.trim() ? suggest(inputEl.value) : (suggestEl.style.display = 'none'));
  inputEl.addEventListener('blur', () => setTimeout(() => suggestEl.style.display = 'none', 150));
  listEl.addEventListener('click', () => inputEl.focus());
  return { add, get: () => tags.slice(), clear: () => { tags.length = 0; render(); } };
}

const tiEventType = makeTagInput(document.getElementById('wqEventTypeTags'), document.getElementById('wqEventTypeInput'), document.getElementById('wqEventTypeSuggest'), allEventTypes);

const tiOrganiser = makeTagInput(document.getElementById('wqOrganiserTags'), document.getElementById('wqOrganiserInput'), document.getElementById('wqOrganiserSuggest'), allOrganisers);
const tiRole      = makeTagInput(document.getElementById('wqRoleTags'),      document.getElementById('wqRoleInput'),      document.getElementById('wqRoleSuggest'),      allRoles);

/* ── Matching ── */
function matches(item, s) {
  if (s.q) {
    const hay = [item.title, item.description, item.location, item.event_type].join(' ').toLowerCase();
    if (!hay.includes(s.q.toLowerCase())) return false;
  }
  if (s.event_types.length && !s.event_types.some(v => (item.event_type||'').toLowerCase().includes(v.toLowerCase()))) return false;
  if (s.locations.length   && !s.locations.some(v   => (item.location||'').toLowerCase().includes(v.toLowerCase())))   return false;
  if (s.organisers.length  && !item.organisers.some(o  => s.organisers.some(v  => o.toLowerCase().includes(v.toLowerCase()))))  return false;
  if (s.roles.length       && !item.sunils_role.some(r => s.roles.some(v       => r.toLowerCase().includes(v.toLowerCase()))))       return false;
  if (s.after  && !(item.date && item.date >= s.after))  return false;
  if (s.before && !(item.date && item.date <= s.before)) return false;
  return true;
}

function getState() {
  return {

    event_types: tiEventType.get(),
    locations:   tiLocation.get(),
    organisers:  tiOrganiser.get(),
    roles:       tiRole.get(),
    after:       document.getElementById('wqAfter').value.trim() || null,
    before:      document.getElementById('wqBefore').value.trim() || null
  };
}

/* ── Sort ── */
let sortCol = 'date', sortDir = -1;

function sorted(items) {
  return items.slice().sort((a, b) => {
    const av = (a[sortCol]||'').toString();
    const bv = (b[sortCol]||'').toString();
    return sortDir * av.localeCompare(bv);
  });
}

function pillsHtml(arr) {
  return (arr||[]).map(v => `<span class="wq-tag-pill">${v}</span>`).join('');
}

function formatDate(d) {
  if (!d) return '—';
  const dt = new Date(d);
  if (isNaN(dt)) return d;
  return dt.toLocaleDateString('en-GB', { day:'numeric', month:'short', year:'numeric' });
}

function render(items) {
  lastResults = items;
  document.getElementById('wqCount').textContent = items.length + (items.length === 1 ? ' result' : ' results');
  const area = document.getElementById('wqResults');

  if (!items.length) {
    area.innerHTML = '<div class="wq-empty"><svg width="40" height="40" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.5"><circle cx="11" cy="11" r="8"/><line x1="21" y1="21" x2="16.65" y2="16.65"/></svg>No results. Try broadening your filters.</div>';
    return;
  }

  const s = sorted(items);

  if (currentView === 'table') {
    const cols = [
      { key:'title',      label:'Event' },
      { key:'event_type', label:'Type' },
      { key:'date',       label:'Date' },
      { key:'location',   label:'Location' },
      { key:'sunils_role',label:'Role' },
    ];
    const head = cols.map(c => {
      const active = sortCol === c.key ? 'sorted' : '';
      const icon = sortCol === c.key ? (sortDir === 1 ? '▲' : '▼') : '⇅';
      return `<th class="${active}" data-col="${c.key}">${c.label} <span class="sort-icon">${icon}</span></th>`;
    }).join('');

    const rows = s.map(it => `
      <tr>
        <td><a href="${it.url}">${it.title}</a></td>
        <td>${it.event_type || '—'}</td>
        <td style="white-space:nowrap">${formatDate(it.date)}</td>
        <td>${it.location || '—'}</td>
        <td>${pillsHtml(it.sunils_role)}</td>
      </tr>`).join('');

    area.innerHTML = `<div class="wq-table-wrap"><table class="wq-table"><thead><tr>${head}</tr></thead><tbody>${rows}</tbody></table></div>`;

    area.querySelectorAll('.wq-table th').forEach(th => {
      th.addEventListener('click', () => {
        if (sortCol === th.dataset.col) sortDir *= -1;
        else { sortCol = th.dataset.col; sortDir = -1; }
        render(lastResults);
      });
    });

  } else {
    const cards = s.map(it => `
      <div class="wq-card">
        <h3><a href="${it.url}">${it.title}</a></h3>
        <div class="wq-card-meta">
          ${it.event_type ? `<span>🏷️ ${it.event_type}</span>` : ''}
          ${it.date       ? `<span>📅 ${formatDate(it.date)}</span>` : ''}
          ${it.location   ? `<span>📍 ${it.location}</span>` : ''}
          ${it.sunils_role && it.sunils_role.length ? `<span>🎭 ${it.sunils_role.join(', ')}</span>` : ''}
          ${it.organisers && it.organisers.length ? `<span>🏛️ ${it.organisers.join(', ')}</span>` : ''}
        </div>
      </div>`).join('');
    area.innerHTML = `<div class="wq-cards">${cards}</div>`;
  }
}

/* ── Run ── */
function run() {
  const s = getState();
  const results = WQ_DATA.filter(d => matches(d, s));
  render(results);
  syncURL(s);
}

/* ── URL sync ── */
function syncURL(s) {
  const p = new URLSearchParams();
  if (s.q) p.set('q', s.q);
  if (s.event_types.length) p.set('et', s.event_types.join('|'));
  if (s.locations.length)   p.set('loc', s.locations.join('|'));
  if (s.organisers.length)  p.set('org', s.organisers.join('|'));
  if (s.roles.length)       p.set('role', s.roles.join('|'));
  if (s.after)  p.set('after', s.after);
  if (s.before) p.set('before', s.before);
  history.replaceState({}, '', location.pathname + '?' + p.toString());
}

function restoreURL() {
  const p = new URLSearchParams(location.search);
  if (p.get('q'))    document.getElementById('wqQ').value = p.get('q');
  if (p.get('after'))  document.getElementById('wqAfter').value = p.get('after');
  if (p.get('before')) document.getElementById('wqBefore').value = p.get('before');
  if (p.get('et'))   p.get('et').split('|').forEach(v => tiEventType.add(v));
  if (p.get('loc'))  p.get('loc').split('|').forEach(v => tiLocation.add(v));
  if (p.get('org'))  p.get('org').split('|').forEach(v => tiOrganiser.add(v));
  if (p.get('role')) p.get('role').split('|').forEach(v => tiRole.add(v));
}

document.getElementById('wqExportCsv').addEventListener('click', () => {
  const items = WQ_DATA.filter(d => matches(d, getState()));
  const keys = ['title','event_type','date','location','sunils_role','organisers','url'];
  const rows = [keys.join(',')].concat(items.map(it => keys.map(k => `"${String(it[k]||'').replace(/"/g,'""')}"`).join(',')));
  const a = document.createElement('a');
  a.href = URL.createObjectURL(new Blob([rows.join('\n')], {type:'text/csv'}));
  a.download = 'events.csv'; a.click();
});
