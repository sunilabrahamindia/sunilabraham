<script>
const DATA = [
{% for p in site.pages %}
{% if p.title and p.created and p.categories %}
{% assign is_event = false %}
{% for cat in p.categories %}
{% if cat == "Events participated" or cat == "Events organised" %}
{% assign is_event = true %}
{% endif %}
{% endfor %}
{% if is_event %}
{
  title: {{ p.title | jsonify }},
  description: {{ p.description | default: "" | jsonify }},
  date: {{ p.date | default: "" | jsonify }},
  created: {{ p.created | jsonify }},
  url: "{{ p.url }}",
  event_type: {{ p.event_type | default: "" | jsonify }},
  location: {{ p.location | default: "" | jsonify }},
  organisers: {{ p.organisers | default: "" | jsonify }},
  sunils_role: {{ p.sunils_role | default: "" | jsonify }},
  website: {{ p.website | default: "" | jsonify }},
  categories: {{ p.categories | default: "" | jsonify }}
},
{% endif %}
{% endif %}
{% endfor %}
];

DATA.forEach(d=>{
  if(typeof d.organisers === 'string') d.organisers = d.organisers ? [d.organisers] : [];
  if(!Array.isArray(d.organisers)) d.organisers = d.organisers ? [d.organisers] : [];
  if(typeof d.sunils_role === 'string') d.sunils_role = d.sunils_role ? [d.sunils_role] : [];
  if(!Array.isArray(d.sunils_role)) d.sunils_role = d.sunils_role ? [d.sunils_role] : [];
  if(typeof d.categories === 'string') d.categories = d.categories ? [d.categories] : [];
});

function uniq(arr){ return [...new Set((arr||[]).filter(Boolean))].sort(); }

const allEventTypes = uniq(DATA.map(d=>d.event_type).filter(Boolean));
const allLocations  = uniq(DATA.map(d=>d.location).filter(Boolean));
const allOrganisers = uniq(DATA.flatMap(d=>d.organisers));
const allRoles      = uniq(DATA.flatMap(d=>d.sunils_role));

function createTagInput(list, tagsContainer, inputEl, suggEl) {
  const state = { tags: [] };
  function renderTags() {
    tagsContainer.innerHTML = '';
    state.tags.forEach((t,i)=>{
      const span = document.createElement('span');
      span.className = 'tag';
      span.innerText = t;
      const btn = document.createElement('button');
      btn.type = 'button';
      btn.setAttribute('aria-label','Remove ' + t);
      btn.innerText = '✕';
      btn.addEventListener('click',()=>{ state.tags.splice(i,1); renderTags(); });
      span.appendChild(btn);
      tagsContainer.appendChild(span);
    });
  }
  function addTag(value) {
    const trimmed = (value||'').trim();
    if(trimmed && !state.tags.includes(trimmed)) { state.tags.push(trimmed); renderTags(); }
  }
  function showSuggestions(q) {
    const ql = (q||'').toLowerCase();
    const candidates = list.filter(x=>x.toLowerCase().includes(ql) && !state.tags.includes(x)).slice(0,25);
    if(!candidates.length){ suggEl.style.display='none'; return; }
    suggEl.innerHTML = '';
    candidates.forEach(c=>{
      const b = document.createElement('button');
      b.type = 'button'; b.textContent = c;
      b.addEventListener('click',()=>{ addTag(c); suggEl.style.display='none'; inputEl.value=''; runQuery(); });
      suggEl.appendChild(b);
    });
    suggEl.style.display = 'block';
  }
  inputEl.addEventListener('keydown',(e)=>{
    if(e.key==='Enter'){ e.preventDefault(); addTag(inputEl.value); inputEl.value=''; suggEl.style.display='none'; runQuery(); }
    if(e.key==='Escape'){ inputEl.value=''; suggEl.style.display='none'; }
  });
  inputEl.addEventListener('input',()=>{ inputEl.value.trim() ? showSuggestions(inputEl.value) : (suggEl.style.display='none'); });
  inputEl.addEventListener('blur',()=>{ setTimeout(()=>suggEl.style.display='none', 120); });
  renderTags();
  return { addTag, getTags:()=>state.tags.slice() };
}

const eventTypeInput = createTagInput(allEventTypes, document.getElementById('eventTypeTags'), document.getElementById('eventTypeInput'), document.getElementById('eventTypeSuggest'));
const locationInput  = createTagInput(allLocations,  document.getElementById('locationTags'),   document.getElementById('locationInput'),   document.getElementById('locationSuggest'));
const organiserInput = createTagInput(allOrganisers, document.getElementById('organiserTags'),  document.getElementById('organiserInput'),  document.getElementById('organiserSuggest'));
const roleInput      = createTagInput(allRoles,      document.getElementById('roleTags'),       document.getElementById('roleInput'),       document.getElementById('roleSuggest'));

function itemMatches(item, state) {
  const qClean = (state.q||'').trim().toLowerCase();
  if(qClean) {
    const hay = [item.title, item.description].join(' ').toLowerCase();
    if(!hay.includes(qClean)) return false;
  }
  if(state.event_types && state.event_types.length)
    if(!state.event_types.some(s=>(item.event_type||'').toLowerCase().includes(s.toLowerCase()))) return false;
  if(state.locations && state.locations.length)
    if(!state.locations.some(s=>(item.location||'').toLowerCase().includes(s.toLowerCase()))) return false;
  if(state.organisers && state.organisers.length)
    if(!item.organisers.some(o=>state.organisers.some(s=>o.toLowerCase().includes(s.toLowerCase())))) return false;
  if(state.roles && state.roles.length)
    if(!item.sunils_role.some(r=>state.roles.some(s=>r.toLowerCase().includes(s.toLowerCase())))) return false;
  if(state.after)  if(!(item.date && item.date >= state.after))  return false;
  if(state.before) if(!(item.date && item.date <= state.before)) return false;
  return true;
}

function sortResults(list, mode) {
  return list.slice().sort((a,b)=>{
    if(mode==='date_desc')  return (b.date||'').localeCompare(a.date||'');
    if(mode==='date_asc')   return (a.date||'').localeCompare(b.date||'');
    if(mode==='title_asc')  return (a.title||'').localeCompare(b.title||'');
    if(mode==='title_desc') return (b.title||'').localeCompare(a.title||'');
    return 0;
  });
}

function renderResults(items) {
  const area = document.getElementById('resultsArea');
  const count = document.getElementById('resultCount');
  area.innerHTML = '';
  if(!items || !items.length){
    area.innerHTML = '<div class="no-results">No results. Try broadening your filters.</div>';
    count.textContent = '0 results';
    return;
  }
  count.textContent = items.length + (items.length===1?' result':' results');
  sortResults(items, document.getElementById('sortSelect').value).forEach(it=>{
    const c = document.createElement('article');
    c.className = 'card';
    c.innerHTML = `
      <h3><a href="${it.url}">${it.title}</a></h3>
      <div class="meta">
        ${it.event_type ? '<div>'+it.event_type+'</div>' : ''}
        ${it.location   ? '<div>📍 '+it.location+'</div>' : ''}
        ${it.date       ? '<div>'+it.date+'</div>' : ''}
        ${it.sunils_role && it.sunils_role.length ? '<div>'+it.sunils_role.join(', ')+'</div>' : ''}
      </div>
      ${it.description ? '<p>'+it.description+'</p>' : ''}
    `;
    area.appendChild(c);
  });
}

function gatherState() {
  return {
    q:           document.getElementById('q').value.trim(),
    event_types: eventTypeInput.getTags(),
    locations:   locationInput.getTags(),
    organisers:  organiserInput.getTags(),
    roles:       roleInput.getTags(),
    after:       document.getElementById('after').value.trim() || null,
    before:      document.getElementById('before').value.trim() || null
  };
}

function runQuery() {
  const state = gatherState();
  renderResults(DATA.filter(d=>itemMatches(d,state)));
  updateURL(state);
}

function updateURL(state) {
  const p = new URLSearchParams();
  if(state.q)           p.set('q', state.q);
  if(state.event_types && state.event_types.length) p.set('event_types', state.event_types.join('|'));
  if(state.locations   && state.locations.length)   p.set('locations',   state.locations.join('|'));
  if(state.organisers  && state.organisers.length)  p.set('organisers',  state.organisers.join('|'));
  if(state.roles       && state.roles.length)       p.set('roles',       state.roles.join('|'));
  if(state.after)  p.set('after',  state.after);
  if(state.before) p.set('before', state.before);
  p.set('sort', document.getElementById('sortSelect').value);
  history.replaceState({}, '', location.pathname + '?' + p.toString());
}

function restoreFromURL() {
  const p = new URLSearchParams(location.search);
  if(p.get('q'))           document.getElementById('q').value = p.get('q');
  if(p.get('after'))       document.getElementById('after').value  = p.get('after');
  if(p.get('before'))      document.getElementById('before').value = p.get('before');
  if(p.get('sort'))        document.getElementById('sortSelect').value = p.get('sort');
  if(p.get('event_types')) p.get('event_types').split('|').forEach(v=>eventTypeInput.addTag(v));
  if(p.get('locations'))   p.get('locations').split('|').forEach(v=>locationInput.addTag(v));
  if(p.get('organisers'))  p.get('organisers').split('|').forEach(v=>organiserInput.addTag(v));
  if(p.get('roles'))       p.get('roles').split('|').forEach(v=>roleInput.addTag(v));
}

document.getElementById('apply').addEventListener('click', runQuery);
document.getElementById('clear').addEventListener('click', ()=>{ history.replaceState({},'',location.pathname); location.reload(); });
document.getElementById('exportJson').addEventListener('click', ()=>{
  const items = DATA.filter(d=>itemMatches(d,gatherState()));
  const a = document.createElement('a');
  a.href = URL.createObjectURL(new Blob([JSON.stringify(items,null,2)],{type:'application/json'}));
  a.download = 'events.json'; a.click();
});
document.getElementById('exportCsv').addEventListener('click', ()=>{
  const items = DATA.filter(d=>itemMatches(d,gatherState()));
  const keys = ['title','event_type','location','date','sunils_role','organisers','url'];
  const rows = [keys.join(',')].concat(items.map(it=>keys.map(k=>`"${String(it[k]||'').replace(/"/g,'""')}"`).join(',')));
  const a = document.createElement('a');
  a.href = URL.createObjectURL(new Blob([rows.join('\n')],{type:'text/csv'}));
  a.download = 'events.csv'; a.click();
});
document.getElementById('sortSelect').addEventListener('change', ()=>{ if(document.querySelectorAll('.card').length) runQuery(); });

restoreFromURL();
</script>
