<h2>Pages created per day (since 19 October)</h2>
<ol id="created-per-day"></ol>

<script>
  // Data injected at build time by Liquid
  const pages = [
    {% for page in site.pages %}
      {% unless page.published == false or page.url contains '/sandbox/' %}
        {% if page.created %}
          {
            url: "{{ page.url }}",
            created: "{{ page.created }}"
          },
        {% endif %}
      {% endunless %}
    {% endfor %}
  ];

  const startDate = "2025-10-19";
  const today = new Date().toISOString().slice(0, 10);

  const counts = {};

  // Count pages by created date
  pages.forEach(p => {
    if (p.created < startDate) return;
    counts[p.created] = (counts[p.created] || 0) + 1;
  });

  // Helper: iterate dates safely (ISO format)
  function nextDate(dateStr) {
    const d = new Date(dateStr);
    d.setDate(d.getDate() + 1);
    return d.toISOString().slice(0, 10);
  }

  const list = document.getElementById("created-per-day");

  // Walk from startDate to today, filling gaps
  for (let d = startDate; d <= today; d = nextDate(d)) {
    const count = counts[d] || 0;
    const label = count === 1 ? "page" : "pages";

    const li = document.createElement("li");
    li.textContent = `${d}: ${count} ${label}`;
    list.appendChild(li);
  }
</script>
