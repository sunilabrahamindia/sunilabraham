<h2>Pages created per day (since 2 December)</h2>
<ol id="created-per-day"></ol>

<script>
  // Data injected at build time by Liquid
  const pages = [
    {% for page in site.pages %}
      {% unless page.published == false or page.url contains '/sandbox/' %}
        {% if page.created %}
          {
            url: "{{ page.url }}",
            created: "{{ page.created }}"
          },
        {% endif %}
      {% endunless %}
    {% endfor %}
  ];

  const startDate = "2025-12-02";
  const counts = {};

  pages.forEach(p => {
    if (p.created < startDate) return;

    if (!counts[p.created]) {
      counts[p.created] = 0;
    }
    counts[p.created]++;
  });

  const list = document.getElementById("created-per-day");

  Object.keys(counts)
    .sort()
    .forEach(date => {
      const li = document.createElement("li");
      li.textContent = `${date}: ${counts[date]} pages`;
      list.appendChild(li);
    });
</script>
