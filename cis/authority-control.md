This page documents authority control identifiers, institutional records, archival references, organisational profiles, and related metadata associated with the [Centre for Internet and Society](/cis/). It is intended to support long-term discoverability, citation consistency, digital preservation, and interoperability across library catalogues, organisational databases, research systems, archival infrastructures, policy networks, and web-based knowledge platforms.

The page consolidates identifiers and external records connected with The Centre for Internet and Society's research publications, policy work, institutional collaborations, conference participation, public advocacy, archival collections, and contributions to internet governance, digital rights, accessibility, technology policy, artificial intelligence governance, and knowledge systems in India and internationally.

As metadata systems frequently fragment institutional identities across abbreviations, domain changes, archived web records, publication databases, and organisational references, this page also serves as a reconciliation layer for authority control and bibliographic standardisation within The Sunil Abraham Project.

<script>
(function () {

  /* ── Search ── */
  var searchInput = document.getElementById('authority-search');
  var clearBtn    = document.getElementById('ac-clear');
  var countEl     = document.getElementById('ac-count');
  var emptyEl     = document.getElementById('ac-empty');
  var items       = document.querySelectorAll('.ac-item');
  var groups      = document.querySelectorAll('.ac-group-card');
  var TOTAL       = items.length;

  function updateCount(n) {
    countEl.textContent = n + (n === 1 ? ' record' : ' records');
  }

  function runSearch(q) {
    var visible = 0;
    items.forEach(function (item) {
      var label = (item.dataset.label || '').toLowerCase();
      var value = (item.dataset.value || '').toLowerCase();
      var text  = item.textContent.toLowerCase();
      var match = !q || label.includes(q) || value.includes(q) || text.includes(q);
      item.setAttribute('aria-hidden', match ? 'false' : 'true');
      if (match) visible++;
    });

    /* Hide groups with no visible items */
    groups.forEach(function (grp) {
      var anyVisible = Array.from(grp.querySelectorAll('.ac-item'))
        .some(function (it) { return it.getAttribute('aria-hidden') !== 'true'; });
      grp.setAttribute('aria-hidden', anyVisible ? 'false' : 'true');
    });

    updateCount(q ? visible : TOTAL);
    emptyEl.classList.toggle('visible', visible === 0 && !!q);
    clearBtn.classList.toggle('visible', !!q);
  }

  searchInput.addEventListener('input', function () {
    runSearch(this.value.toLowerCase().trim());
  });

  clearBtn.addEventListener('click', function () {
    searchInput.value = '';
    runSearch('');
    searchInput.focus();
  });

  /* Keyboard: Escape clears */
  searchInput.addEventListener('keydown', function (e) {
    if (e.key === 'Escape' && this.value) {
      this.value = '';
      runSearch('');
    }
  });

  updateCount(TOTAL);
})();
</script>
